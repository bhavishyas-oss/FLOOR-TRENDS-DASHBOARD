/* ═══════════════════════════════════════════════ HELPERS ═══════════════════════════════════════════════ */ function normalize(val){ return String(val || '').trim().toUpperCase(); } function isPivotFmt(headers){ const h = headers.map(x => normalize(x)); return h.includes('GRAND TOTAL') || h.includes('TOTAL'); } /* ═══════════════════════════════════════════════ FILE TYPE IDENTIFICATION ═══════════════════════════════════════════════ */ function identifyFileType(fileName, headers){ const name = fileName.toLowerCase(); const h = headers.map(x => normalize(x)); if(name.includes('lmodc') || h.includes('LMODC')) return 'lmodc'; if(name.includes('short') || h.includes('SHORT SHIPMENT')) return 'forwardShort'; if(h.includes('REASON OF LOSS')) return 'reverse'; if(h.includes('PIVOT') || h.includes('RECEIVED AT DC')) return 'forward'; throw new Error('Unknown file type'); } /* ═══════════════════════════════════════════════ RAW ROW PROCESSING ═══════════════════════════════════════════════ */ function dispatchRaw(type,row,date,agg){ const awb = normalize(row['AWB'] || row['awb'] || row['tracking']); if(!awb) return; let bucket = 'D0'; if(type === 'forward'){ const pivot = normalize(row['Pivot']); if(!pivot.includes('RECEIVED AT DC')) return; bucket = normalize(row['Aging'] || row['Bucket'] || 'D0'); } if(type === 'reverse'){ const reason = normalize(row['Reason of Loss']); if(!['RECEIVED AT DC','MISROUTED'].includes(reason)) return; bucket = normalize(row['Aging'] || 'D0'); } if(type === 'lmodc' || type === 'forwardShort'){ bucket = normalize(row['Aging'] || row['Bucket'] || 'D0'); } const key = `${awb}_${bucket}_${date}`; if(!agg[key]){ agg[key] = { awb, bucket, date, count: 0 }; } agg[key].count++; } /* ═══════════════════════════════════════════════ PIVOT PROCESSING (LMODC + SHORT ONLY) ═══════════════════════════════════════════════ */ function processPivot(rows,type,date,headers,agg){ for(const row of rows){ const awb = normalize(row['AWB'] || row[headers[0]]); if(!awb) continue; for(const col of headers){ const bucket = normalize(col); if(bucket.startsWith('D')){ const val = Number(row[col]) || 0; if(val > 0){ const key = `${awb}_${bucket}_${date}`; agg[key] = { awb, bucket, date, count: val }; } } } } } /* ═══════════════════════════════════════════════ APPLY ROWS (FINAL FIXED VERSION) ═══════════════════════════════════════════════ */ function applyRows(rows,type,date,headers){ const pivotFormat = isPivotFmt(headers); const agg = {}; if(pivotFormat){ // ✅ Only allow pivot for LMODC + Short Shipment if(type === 'lmodc' || type === 'forwardShort'){ processPivot(rows,type,date,headers,agg); } else { return { awbs:0, pivot:true, records:0, skipped:true }; } } else { for(const row of rows){ if(!Object.values(row).some(v => String(v).trim())) continue; dispatchRaw(type,row,date,agg); } } const records = Object.values(agg); master[type] = (master[type] || []).filter(r => r.date !== date); master[type].push(...records); const awbs = records.reduce((sum,r) => sum + r.count, 0); // ✅ DEBUG (keeps you sane) console.log(`[${type}] processed`, { rows: rows.length, output: records.length, awbs, pivot: pivotFormat }); return { awbs, pivot: pivotFormat, records: records.length }; }
